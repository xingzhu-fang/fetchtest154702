name: fetch
on:
  workflow_dispatch:
  push:
    paths: [urls.txt,.github/workflows/fetch.yml]
permissions:
  contents: write
jobs:
  fetch:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Fetch pages
        run: |
          mkdir -p pages
          python - <<'PY'
          import requests,os
          urls=[l.strip() for l in open('urls.txt') if l.strip() and not l.startswith('#')]
          for i,url in enumerate(urls):
            fn='pages/%03d.html'%i
            try:
              r=requests.get(url,timeout=40,headers={'User-Agent':'Mozilla/5.0 (research)'} )
              open(fn,'wb').write(r.content)
              open(fn+'.meta','w').write(f'{url}\nstatus {r.status_code}\nfinal {r.url}\n')
              print(i,url,r.status_code,len(r.content))
            except Exception as e:
              open(fn+'.meta','w').write(f'{url}\nerror {e}\n');print(i,e)
          PY
      - name: commit
        run: |
          git config user.email 'bot@users.noreply.github.com'; git config user.name bot
          git add pages || true
          git commit -m fetched || true
          git push || true
